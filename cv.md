function combine(...objects) {
  return objects.reduce((acc, obj) => {
    for (const key in obj) {
      if (obj.hasOwnProperty(key)) {
        acc[key] ? (acc[key] += obj[key]) : (acc[key] = obj[key])
      }
    }
    return acc
  }, {})
}
